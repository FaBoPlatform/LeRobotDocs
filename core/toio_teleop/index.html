from typing import Any, Optional
import math

from lerobot.teleoperators.teleoperator import Teleoperator
from .config_joystick import ToioConfig

# 速度スケール（[-1..1] の軸値に掛ける）
SPEED = 0.5
# デッドゾーン（小さなガタつきを無視）
DEADZONE = 0.08

# 参照スクリプトにならい、既定は X=axis(0), Y=axis(2)
DEFAULT_AXIS_X = 0
DEFAULT_AXIS_Y = 2
# 多くのパッドでは上に倒すと Y がマイナスになるため反転して上を + に
INVERT_Y = True
INVERT_X = True 


class Toio(Teleoperator):
    """
    pygame を用いた最小ジョイスティック teleop
      - 左/右スティックの軸値（既定: X=axis0, Y=axis2）を読み取り、毎フレーム {vx, vy} を返す
      - フィードバックは受け取らない
    """
    config_class = ToioConfig
    name = "toio"

    def __init__(self, config: ToioConfig):
        super().__init__(config)
        self.config = config

        self._connected = False
        self._calibrated = True

        self._vx: float = 0.0
        self._vy: float = 0.0

        # pygame / joystick
        self._pg = None                    # lazy import: pygame モジュール
        self._joy = None                   # pygame.joystick.Joystick インスタンス
        self._warned_no_joy = False
        self._warned_axis = False

        # 軸インデックス（config 側で上書きできるようにフォールバック付き）
        self._axis_x_idx = int(getattr(config, "axis_x_index", DEFAULT_AXIS_X))
        self._axis_y_idx = int(getattr(config, "axis_y_index", DEFAULT_AXIS_Y))
        self._invert_x = bool(getattr(config, "invert_x", INVERT_X))
        self._invert_y = bool(getattr(config, "invert_y", INVERT_Y))

        # スケールとデッドゾーン（config から上書き可能）
        self._speed = float(getattr(config, "speed", SPEED))
        self._deadzone = float(getattr(config, "deadzone", DEADZONE))

    # ----- Teleoperator interface -----
    @property
    def action_features(self) -> dict[str, type]:
        return {"vx": float, "vy": float}

    @property
    def feedback_features(self) -> dict[str, type]:
        return {}

    @property
    def is_connected(self) -> bool:
        return self._connected

    def connect(self, calibrate: bool = True) -> None:
        # pygame はここで初期化（エラーはわかりやすく）
        try:
            import pygame  # type: ignore
        except Exception as e:
            raise ImportError(
                "pygame が見つかりません。`pip install pygame` を実行してください。"
            ) from e

        self._pg = pygame
        pygame.init()
        pygame.joystick.init()

        cnt = pygame.joystick.get_count()
        if cnt <= 0:
            print("[teleop/joystick] No joystick detected (vx, vy は 0 のままになります)")
            self._joy = None
        else:
            self._joy = pygame.joystick.Joystick(0)
            self._joy.init()
            print(f"[teleop/joystick] Connected to: {self._joy.get_name()} (axes={self._joy.get_numaxes()})")

        self._connected = True

    @property
    def is_calibrated(self) -> bool:
        return self._calibrated

    def calibrate(self) -> None:
        pass

    def configure(self) -> None:
        pass

    def disconnect(self) -> None:
        self._connected = False
        try:
            if self._joy is not None:
                self._joy.quit()
        except Exception:
            pass
        try:
            if self._pg is not None:
                self._pg.joystick.quit()
                self._pg.quit()
        except Exception:
            pass
        self._joy = None
        self._pg = None

    # ----- 内部ユーティリティ -----
    def _apply_deadzone_and_scale(self, v: float) -> float:
        # デッドゾーン: 原点付近は 0 に
        if abs(v) < self._deadzone:
            v = 0.0
        # スケール
        return float(self._speed * v)

    def _read_axes(self) -> None:
        if self._pg is None:
            return

        # イベントを処理（Queue を空に）
        self._pg.event.pump()

        if self._joy is None:
            # ジョイスティックが無い場合は 0 を維持
            self._vx, self._vy = 0.0, 0.0
            if not self._warned_no_joy:
                print("[teleop/joystick] Warning: joystick not available (vx,vy=0)")
                self._warned_no_joy = True
            return

        num_axes = self._joy.get_numaxes()

        # 軸インデックス安全化（足りない場合は Y=1 にフォールバック）
        ax_x = self._axis_x_idx
        ax_y = self._axis_y_idx
        if num_axes <= max(ax_x, ax_y):
            if num_axes >= 2:
                ax_y = 1  # よくある配置: 0=X, 1=Y
            else:
                # 軸が 1 本以下なら 0 固定
                self._vx, self._vy = 0.0, 0.0
                if not self._warned_axis:
                    print(f"[teleop/joystick] Warning: not enough axes (have {num_axes})")
                    self._warned_axis = True
                return

        # 軸値の取得（-1..1）
        try:
            x = float(self._joy.get_axis(ax_x))
            y = float(self._joy.get_axis(ax_y))
        except Exception:
            x, y = 0.0, 0.0

        # 反転指定（一般に Y は上がマイナスなので反転して + にする）
        if self._invert_x:
            x = -x
        if self._invert_y:
            y = -y

        # デッドゾーン＆スケール
        self._vx = self._apply_deadzone_and_scale(x)
        self._vy = self._apply_deadzone_and_scale(y)

    # ----- 出力（毎フレーム呼ばれる） -----
    def get_action(self) -> dict[str, Any]:
        self._read_axes()
        # UDP 送信は不要：ここで {vx,vy} を返せばループがそのまま Robot に渡す
        return {"vx": float(self._vx), "vy": float(self._vy)}

    def send_feedback(self, feedback: dict[str, float]) -> None:
        # フィードバックは扱わない
        pass
